Change a parameter when a filter changesÔÉÅ

Use get_parameter() inside the callback function and change the value using simple assignment (param.value = 15).

 1 from ._anvil_designer import MainTemplate
 2 from anvil import *
 3 
 4 from tableau_extension import api
 5 
 6 class Main(MainTemplate):
 7   def __init__(self, **properties):
 8     self.init_components(**properties)
 9 
10     self.dashboard = api.get_dashboard()
11     self.dashboard.register_event_handler('filter_changed', self.modify_param)
12 
13   def modify_param(self, event):
14     """ This method adds one to a parameter whenever a filter is changed. """
15     param = self.dashboard.get_parameter('one')
16     param.value += 1
17 
18   def primary_button_1_click(self, **event_args):
19     """This method is called when the button is clicked"""
20     param = self.dashboard.get_parameter('one')
21     param.value = 500
